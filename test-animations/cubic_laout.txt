    <script is:inline>
      // Play the diagonal bars + logo intro animation.
      // If a callback is provided, it runs after the animation finishes.
      const playIntroOverlay = (callback) => {
        const overlay = document.getElementById('intro-overlay');
        if (!overlay) {
          if (callback) callback();
          return;
        }

        const center = overlay.querySelector('.intro-center');
        const animatedElements = overlay.querySelectorAll('.intro-bar, .intro-center');

        // Unhide overlay
        overlay.classList.remove('intro-overlay-hidden');

        // Reset animations so they can replay
        animatedElements.forEach((el) => {
          el.classList.add('intro-reset');
          // force reflow
          void el.offsetWidth;
          el.classList.remove('intro-reset');
        });

        const finish = () => {
          overlay.classList.add('intro-overlay-hidden');
          if (callback) callback();
        };

        if (center) {
          center.addEventListener('animationend', finish, { once: true });
          // Fallback
          setTimeout(finish, 2000);
        } else {
          setTimeout(finish, 2000);
        }
      };

      const showPage = () => {
        const main = document.getElementById('page-root');
        if (!main) return;
        main.classList.remove('opacity-0');
        main.classList.remove('page-fade-out');
        main.classList.add('page-fade-in');
      };

      const hidePageWithFade = (onHidden) => {
        const main = document.getElementById('page-root');
        if (!main) {
          if (onHidden) onHidden();
          return;
        }
        main.classList.remove('page-fade-in');
        main.classList.add('page-fade-out');

        const done = () => {
          main.removeEventListener('animationend', done);
          if (onHidden) onHidden();
        };

        main.addEventListener('animationend', done, { once: true });
        // Fallback
        setTimeout(done, 400);
      };

      const enablePageTransitions = () => {
        const links = document.querySelectorAll('a[data-page-link]');
        links.forEach((link) => {
          link.addEventListener('click', (event) => {
            const href = link.getAttribute('href');
            if (!href) return;

            const url = new URL(href, window.location.origin);
            if (url.origin !== window.location.origin || url.pathname === window.location.pathname) {
              return;
            }

            event.preventDefault();
            hidePageWithFade(() => {
              window.location.href = href;
            });
          });
        });
      };

      const init = () => {
        // Always show bars first on load, then fade the page in
        playIntroOverlay(() => {
          showPage();
          enablePageTransitions();
        });
      };

      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
      } else {
        init();
      }
    </script>
